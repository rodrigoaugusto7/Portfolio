def density_plot_var():
    # - Desenha Plot
    plt.figure(figsize=(16, 10), dpi=80)
    try:
        sns.kdeplot(df_lifetime_rfm.loc[df_lifetime_rfm['categoria'] == menu_distri.value,
                           menu_rfm.value], shade=True, color=plt.cm.Spectral(300), label="Distribuio", alpha=.7, shade_lowest=True)
    except:
        sns.kdeplot(df_lifetime_rfm.replace([np.inf, -np.inf], np.nan).dropna().loc[df_lifetime_rfm['categoria'] == menu_distri.value,
                                                                       menu_rfm.value], shade=True, color="dodgerblue", label="Distribuio", alpha=.7)

    # - Decorao
    plt.title('Distribuio '+menu_distri.value, fontsize=22)
    plt.legend()
    plt.show()
    return
